## 建表语句
```sql
CREATE TABLE PIMENTITY
(
  ID int PRIMARY KEY AUTO_INCREMENT,
  priority VARCHAR(20) DEFAULT 'normal',
  todo_ID int,
  note_ID int,
  appointment_ID int,
  contact_ID int,
  CONSTRAINT `e_todo` FOREIGN KEY(todo_ID) REFERENCES PIMTODO(ID),
  CONSTRAINT `e_note` FOREIGN KEY(note_ID) REFERENCES PIMNOTE(ID),
  CONSTRAINT `e_appointment` FOREIGN KEY(appointment_ID) REFERENCES PIMAPPOINTMENT(ID),
  CONSTRAINT `e_contact` FOREIGN KEY (contact_ID) REFERENCES PIMCAONTACT(ID)
)ENGINE Innodb CHARSET utf8;

CREATE TABLE PIMTODO
(
  `ID` INT PRIMARY KEY AUTO_INCREMENT,
  `date` TIMESTAMP DEFAULT now(),
  `item` TEXT DEFAULT ''
)ENGINE Innodb CHARSET utf8;

CREATE TABLE PIMNOTE
(
  `ID` INT PRIMARY KEY AUTO_INCREMENT,
  `text`  TEXT DEFAULT ''
)ENGINE Innodb CHARSET utf8;

CREATE TABLE PIMAPPOINTMENT
(
  `ID` INT PRIMARY KEY AUTO_INCREMENT,
  `date` TIMESTAMP DEFAULT now(),
  `discription` TEXT DEFAULT ''
)ENGINE Innodb CHARSET utf8;

CREATE TABLE PIMCONTACT
(
  `ID` INT PRIMARY KEY AUTO_INCREMENT,
  `FIRST_NAME` VARCHAR(10) DEFAULT NULL ,
  `LAST_NAME` VARCHAR(20) DEFAULT NOT NULL ,
  `EMAIL_ADDRESS` VARCHAR(30) DEFAULT NULL 
)ENGINE Innodb CHARSET utf8;
```
